import timeit

test1 = """
users = {'a':'active', 'b':'inactive', 'c':'active'}

def dict_copy(input_dict: dict) -> None:
  for user, status in input_dict.copy().items():
    if status == 'inactive':
      del users[user]
"""

test2 = """
users = {'a':'active', 'b':'inactive', 'c':'active'}

def dict_items(input_dict: dict) -> None:
  for user, status in input_dict.items():
    if status == 'inactive':
      del [users][user]
"""


print(
  timeit.timeit(
    setup=test1
  )
)

print(
  timeit.timeit(
    setup=test2
  )
)
