// main.cpp

#include <string>
#include <iostream>
#include <typeinfo>
#include <cstdint> // Add this line to include the <cstdint> header file

using namespace std;

int main()
{
    
    // declare char
    char x = 'a';
    cout << typeid(x).name() << endl;

    // auto cast variable 'y' as a pointer to 'x'
    // where '&x' refers to the memory address of 'x'
    auto y = &x;
    // print the type of variable 'y'
    cout << typeid(y).name() << endl;

    // declare void pointer 'z'
    void* z;
    cout << typeid(z).name() << endl;

    // declare int pointer 'i' with no initial value
    int* i;
    cout << typeid(i).name() << endl;

    // declare char pointer 'p'
    // where '&x' refers to the memory address of 'x'
    char* p = &x;

    // dereference pointer
    cout << *p << endl;
    // get address of pointer
    cout << &p << endl;

    // manually cast pointer to long
    // not recommended b/c it's error prone
    uintptr_t addr = 0x7ffddbffaa38;
    long* v = reinterpret_cast(addr);
    cout << typeid(v).name() << endl;

    return 0;

};
